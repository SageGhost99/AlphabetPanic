    function shuffle(arr) {

      const copy = arr.slice();

      for (let i = copy.length - 1; i > 0; i--) {

        const j = Math.floor(Math.random() * (i + 1));

        [copy[i], copy[j]] = [copy[j], copy[i]];

      }

      return copy;

    }

    function startGame(idx) {

      game.category = game.options[idx];

      game.time = 60;

    }


    function startTimer() {

      if (game.timer) clearInterval(game.timer);

      game.timer = setInterval(() => {

        if (!game.active || game.time <= 0) {

          clearInterval(game.timer);

          return;

        }

        game.time -= 0.1;

        if (game.time <= 0) {

          game.time = 0;

          game.active = false;

          game.end = Date.now();

          game.screen = 'end';

          render();

        } else {

          updateTimer();

        }

      }, 100);

    }


    function updateTimer() {

      }

    }


    function clickLetter(letter) {

      if (!game.active || game.clicked.has(letter)) return;

      game.clicked.add(letter);

      game.time += 3;

      

      const btn = document.getElementById('btn-' + letter);

      if (btn) {

        btn.className = 'aspect-square text-3xl font-bold rounded-xl transition-all transform bg-green-500 text-white scale-95 cursor-not-allowed';

        btn.disabled = true;

      }

      

      const counter = document.getElementById('counter');

      if (counter) {

        counter.textContent = 'Letters Clicked: ' + game.clicked.size + ' / 26';

      }

      

      if (game.clicked.size === 26) {

        game.active = false;

        game.end = Date.now();

        game.screen = 'end';

        clearInterval(game.timer);

        render();

      }

    }
